# File Window_Message.rb, line 128
  def fiber_main
  $game_message.visible = true
  update_background
  update_placement
  loop do
    process_all_text if $game_message.has_text?
    process_input
    $game_message.clear
    @gold_window.close
    Fiber.yield
    break unless text_continue?
  end
  close_and_wait
  $game_message.visible = false
  @fiber = nil
end

Source: show 

# File Window_Message.rb, line 260
  def process_escape_character(code, text, pos)
  case code.upcase
  when '$'
    @gold_window.open
  when '.'
    wait(15)
  when '|'
    wait(60)
  when '!'
    input_pause
  when '>'
    @line_show_fast = true
  when '<'
    @line_show_fast = false
  when '^'
    @pause_skip = true
  else
    super
  end
end

  end
end
