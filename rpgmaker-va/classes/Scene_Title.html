  def draw_game_title
  @foreground_sprite.bitmap.font.size = 48
  rect = Rect.new(0, 0, Graphics.width, Graphics.height / 2)
  @foreground_sprite.bitmap.draw_text(rect, $data_system.game_title, 1)
end
