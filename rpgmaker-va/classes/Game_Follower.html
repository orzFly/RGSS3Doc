  def chase_preceding_character
  unless moving?
    sx = distance_x_from(@preceding_character.x)
    sy = distance_y_from(@preceding_character.y)
    if sx != 0 && sy != 0
      move_diagonal(sx > 0 ? 4 : 6, sy > 0 ? 8 : 2)
    elsif sx != 0
      move_straight(sx > 0 ? 4 : 6)
    elsif sy != 0
      move_straight(sy > 0 ? 8 : 2)
    end
  end
end
