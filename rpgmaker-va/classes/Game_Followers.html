  def initialize(leader)
  @visible = $data_system.opt_followers
  @gathering = false                    # 集合处理中的标志
  @data = []
  @data.push(Game_Follower.new(1, leader))
  (2...$game_party.max_battle_members).each do |index|
    @data.push(Game_Follower.new(index, @data[-1]))
  end
end
